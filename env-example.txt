# Puerto de la base de datos
DB_PORT=3306

# Usuario de la base de datos

# Contraseña de la base de datos
DB_PASSWORD=your_password_here

# Nombre de la base de datos

# CONFIGURACIÓN DE LA APLICACIÓN
# ============================================

# Puerto en el que corre la aplicación
PORT=3000

# Entorno de ejecución (development | production)
NODE_ENV=development

# ============================================
# CONFIGURACIÓN DE JWT ✅ (REQUERIDO)
# ============================================

# Secret key para firmar los tokens JWT
# ⚠️ IMPORTANTE: Esta variable es REQUERIDA. La aplicación NO iniciará sin ella.
# Genera una clave segura con: node -e "console.log(require('crypto').randomBytes(64).toString('hex'))"
# O ejecuta: npm run generate:keys
JWT_SECRET=your_super_secret_jwt_key_change_this_in_production_minimum_32_characters

# Tiempo de expiración del token (ej: 1h, 24h, 7d)
# ⚠️ IMPORTANTE: Esta variable es REQUERIDA.
JWT_EXPIRATION=24h

# Tiempo de extensión para renovación automática (ej: 10m, 30m, 1h)
# Cuando se renueva un token, se extiende por este tiempo adicional
